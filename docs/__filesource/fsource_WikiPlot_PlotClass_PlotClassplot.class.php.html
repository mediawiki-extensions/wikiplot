  241.     function GetHash()
  242.     {
  243.         $Hash  = "C:" . $this->Caption;
  244.         $Hash .= "F:" . $this->CaptionFont;
  245.         $Hash .= "W:" . $this->Width;
  246.         $Hash .= "H:" . $this->Height;
  247.         $Hash .= "X:" . $this->MinX . "_" . $this->MaxX;
  248.         $Hash .= "Y:" . $this->MinY . "_" . $this->MaxY;
  249.         $Hash .= "A:" . $this->EnableAxis;
  250.         $Hash .= "G:" . $this->EnableGrid . "_" . $this->GridColor . "_" . $this->GridFont;
  251.         $Hash .= "S:" . $this->XGridSpace . "_" . $this->YGridSpace;
  252.         $Hash .= "V:" . "$LastChangedRevision$";
  253.         foreach($this->Graphs as $key => $S)
  254.         {
  255.             $Hash .= "G:" . $key. "_" . $S->GetHash(); 
  256.         }
  257.         return md5($Hash);
  258.     }

  285.     *@return ImageResource 
  286.     */
  287.     function DrawPlot()
  288.     {
  289.         //Get ImageResource
  290.         $ImageResource = imagecreatetruecolor($this->Height,$this->Width);
  291.  
  292.         //AntiAlias ON
  293.         imageantialias($ImageResource,true);
  294.  
  295.         //Get a black Color
  296.         $Black = imagecolorexact($ImageResource,0,0,0);
  297.  
  298.         //Fill the image with white
  299.         imagefill($ImageResource,0,0,imagecolorexact($ImageResource,255,255,255));
  300.  
  301.         //If grid is enabled
  302.         if($this->EnableGrid)
  303.         {
  304.             $this->DrawGrid($ImageResource);
  305.         }
  306.  
  307.         //If axis is enabled
  308.         if($this->EnableAxis)
  309.         {
  310.             $this->DrawAxis($ImageResource);
  311.         }
  312.         
  313.         //Draw caption
  314.         $this->DrawCaption($ImageResource);
  315.  
  316.         //Y position for Labels relative to Image
  317.         $LabelY = 5;
  318.  
  319.         //Plot all graphs
  320.         foreach($this->Graphs as $key => $S)
  321.         {
  322.             //Get Color
  323.             $Color = imagecolorexact($ImageResource,$S->Color[0],$S->Color[1],$S->Color[3]);
  324.  
  325.             //Set Expression
  326.             $m = new EvalMath;
  327.             $m->evaluate("f(x) = " . $S->Exp);
  328.  
  329.             //Set OldCoordinat*, don't start with a line from 0,0
  330.             $OldCoordinatX = $this->GetCoordinatX(0);
  331.             $OldCoordinatY = $m->evaluate("f(".$OldCoordinatX.")");
  332.  
  333.             //Plot the graph
  334.             for($ImageX=0;$ImageX<$this->Width;$ImageX++)
  335.             {
  336.  
  337.                 //Get some NewCoordinat*
  338.                 $NewCoordinatX = $this->GetCoordinatX($ImageX);
  339.                 $NewCoordinatY = $m->evaluate("f(".$NewCoordinatX.") ");
  340.  
  341.                 //Draw a line from OldCoordinat*
  342.                 imageline(
  343.                     $ImageResource,
  344.                     $this->GetImageX($OldCoordinatX),
  345.                     $this->GetImageY($OldCoordinatY),
  346.                     $this->GetImageX($NewCoordinatX),
  347.                     $this->GetImageY($NewCoordinatY),
  348.                     $Color);
  349.  
  350.                 //Get some OldCoordinat*
  351.                 $OldCoordinatX = $NewCoordinatX;
  352.                 $OldCoordinatY = $NewCoordinatY;
  353.             }
  354.  
  355.             //Draw label if it is enabled
  356.             if($S->EnableLabel)
  357.             {
  358.                 //Draw label
  359.                 imagestring($ImageResource,$S->LabelFont,5,$LabelY,"- ".$S->Label,$Color);
  360.  
  361.                 //Add Label height to next Label X position
  362.                 $LabelY += imagefontheight($S->LabelFont);
  363.             }
  364.         }
  365.  
  366.         //Return ImageResource
  367.         return $ImageResource;
  368.     }

  386.     //Rewrite numbers into something short 1.000.000 -> 1E6
  387.         function ShortNumber($Number, $MaxLen = 7)
  388.     {
  389.         //If $Number isn't too long return it as it is
  390.         if(strlen($Number)<=$MaxLen)
  391.         {
  392.             return $Number;
  393.         }else{
  394.             //Convert to scientific notation
  395.             $NSci = sprintf("%e",$Number);
  396.  
  397.             //Follwing hack prevents the function from showing too many decimals
  398.             $ArrNSci = split($NSci,"e");
  399.             return round($ArrNSci[0],$MaxLen-5) . "e" . $ArrNSci[1];
  400.         }
  401.     }

  442.     //Draw X grid
  443.         function DrawXGrid(&$ImageResource)
  444.     {
  445.         //Get grid width
  446.         $XGridSpace = $this->GetXGridSpace();
  447.         
  448.         //Get color to draw with
  449.         $Color = imagecolorexact($ImageResource,$this->GridColor[0],$this->GridColor[1],$this->GridColor[2]);
  450.         //Get text color
  451.         $Black = imagecolorexact($ImageResource,0,0,0);
  452.  
  453.         //Calculate start and end coordinats of the grid
  454.         $XGridStart = ($this->MinX-fmod($this->MinX,$XGridSpace));
  455.         $XGridEnd = $this->MaxX-fmod(($this->MaxX-$this->MinX),$XGridSpace);
  456.  
  457.         //Draw the grid
  458.         for($XCordinate=$XGridStart;$XCordinate<$XGridEnd;$XCordinate+=$XGridSpace)
  459.         {
  460.             imageline(
  461.                 $ImageResource,
  462.                 $this->GetImageX($XCordinate),
  463.                 $this->GetImageY($this->MinY),
  464.                 $this->GetImageX($XCordinate),
  465.                 $this->GetImageY($this->MaxY),
  466.                 $Color);
  467.  
  468.             //If Y axes is not on the image (working in ImageSpace not CoordinatSpace)
  469.             $Y = $this->GetImageY(0);
  470.             if($Y > ($this->Height-imagefontheight($this->GridFont)))
  471.             {
  472.                 $Y = $this->Height-(imagefontheight($this->GridFont)+2);
  473.             }else{
  474.                 if($Y<0)
  475.                 {
  476.                     $Y = 0;
  477.                 }
  478.             }
  479.             imagestring(
  480.                 $ImageResource,
  481.                 $this->GridFont,
