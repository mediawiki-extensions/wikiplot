  260.     function GetHash()
  261.     {

  263.         $Hash .= "F:" . $this->CaptionFont;
  264.         $Hash .= "W:" . $this->Width;

  266.         $Hash .= "X:" . $this->MinX . "_" . $this->MaxX;
  267.         $Hash .= "Y:" . $this->MinY . "_" . $this->MaxY;
  268.         $Hash .= "A:" . $this->EnableAxis;
  269.         $Hash .= "G:" . $this->EnableGrid . "_" . $this->GridColor . "_" . $this->GridFont;
  270.         $Hash .= "S:" . $this->XGridSpace . "_" . $this->YGridSpace;
  271.         $Hash .= "V:" . "$LastChangedRevision: 63 $";
  272.         foreach($this->Graphs as $key => $S)
  273.         {
  274.             $Hash .= "G:" . $key. "_" . $S->GetHash(); 
  275.         }
  276.         return md5($Hash);
  277.     }

  279.     /**
  280.     *Get ImageResource of the plot
  281.     *
  282.     *Generates ImageResource representation of the plot.
  283.     *
  284.     *@access public
  285.     *@uses EnableGrid
  286.     *@uses DrawGrid()
  287.     *@uses $Width
  288.     *@uses $Height
  289.     *@uses $EnableAxis
  290.     *@uses DrawAxis()
  291.     *@uses DrawCaption()
  292.     *@uses DrawPlots()
  293.     *@uses $BackgroundColor
  294.     *@param ImageResource $ImageResource Defaults to null, will generate empty ImageResource.
  295.     *@param Boolean $ChangeSize May we change the size of the plot to fit given ImageResource?
  296.     *@return ImageResource ImageResource representation of the plot.
  297.     */
  298.     function GeneratePlot($ImageResource = null, $ChangeSize = false)
  299.     {
  300.         //If ImageResource is null
  301.         if(is_null($ImageResource))
  302.         {
  303.             //Get ImageResource
  304.             $ImageResource = imagecreatetruecolor($this->Height,$this->Width);
  305.  
  306.             //AntiAlias ON
  307.             imageantialias($ImageResource,true);
  308.  
  309.             //Fill the image with white
  310.             imagefill($ImageResource,0,0,imagecolorexact($ImageResource,$this->BackgroundColor[0],$this->BackgroundColor[1],$this->BackgroundColor[2]));
  311.         
  312.         }//If ImageResource doesn't fit image and we may not change size
  313.         elseif($ChangeSize==false&&(imagesx($ImageResource)!=$this->Width||imagesy($ImageResource)!=$this->Height))
  314.         {
  315.             //Get ImageResource
  316.             $ImageResource = imagecreatetruecolor($this->Height,$this->Width);
  317.  
  318.             //AntiAlias ON
  319.             imageantialias($ImageResource,true);
  320.  
  321.             //Fill the image with white
  322.             imagefill($ImageResource,0,0,imagecolorexact($ImageResource,$this->BackgroundColor[0],$this->BackgroundColor[1],$this->BackgroundColor[2]));            
  323.         }//If we may change the size of the plot
  324.         elseif($ChangeSize)
  325.         {
  326.             //Changing size of the plot.
  327.             $this->Width = imagesx($ImageResource);
  328.             $this->Height = imagesy($ImageResource);
  329.         }
  330.  
  331.         //If grid is enabled
  332.         if($this->EnableGrid)
  333.         {
  334.             $this->DrawGrid($ImageResource);
  335.         }
  336.  
  337.         //If axis is enabled
  338.         if($this->EnableAxis)
  339.         {
  340.             $this->DrawAxis($ImageResource);
  341.         }
  342.         
  343.         //Draw caption
  344.         $this->DrawCaption($ImageResource);
  345.  
  346.         //Draw plots
  347.         $this->DrawPlots($ImageResource);
  348.  
  349.         //Return ImageResource
  350.         return $ImageResource;
  351.     }

  353.     /**
  354.     *Get ImageResource of the plot
  355.     *
  356.     *Generates ImageResource representation of the plot.
  357.     *
  358.     *@access private
  359.     *@uses $Width
  360.     *@uses EvalMath
  361.     *@uses EvalMath::evaluate()
  362.     *@uses GetCoordinatX()
  363.     *@uses GetImageX()
  364.     *@uses GetImageY()
  365.     *@uses $Graphs
  366.     *@uses Graph::$Color
  367.     *@uses Graph::$LabelFont
  368.     *@uses Graph::$EnableLabel
  369.     *@uses Graph::$Label
  370.     *@param ImageResource &$ImageResource ImageResource representation of the plot.
  371.     */
  372.     function DrawPlots(&$ImageResource)
  373.     {
  374.         //Get a black Color
  375.         $Black = imagecolorexact($ImageResource,0,0,0);
  376.  
  377.         //Y position for Labels relative to Image
  378.         $LabelY = 5;
  379.  
  380.         //Plot all graphs
  381.         foreach($this->Graphs as $key => $S)
  382.         {
  383.             //Get Color
  384.             $Color = imagecolorexact($ImageResource,$S->Color[0],$S->Color[1],$S->Color[3]);
  385.  
  386.             //Set Expression
  387.             $m = new EvalMath;
  388.             $m->evaluate("f(x) = " . $S->Exp);
  389.  
  390.             //Set OldCoordinat*, don't start with a line from 0,0
  391.             $OldCoordinatX = $this->GetCoordinatX(0);
  392.             $OldCoordinatY = $m->evaluate("f(".$OldCoordinatX.")");
  393.  
  394.             //Plot the graph
  395.             for($ImageX=0;$ImageX<$this->Width;$ImageX++)
  396.             {
  397.  
  398.                 //Get some NewCoordinat*
  399.                 $NewCoordinatX = $this->GetCoordinatX($ImageX);
  400.                 $NewCoordinatY = $m->evaluate("f(".$NewCoordinatX.") ");
  401.  
  402.                 //Draw a line from OldCoordinat*
  403.                 imageline(
  404.                     $ImageResource,
  405.                     $this->GetImageX($OldCoordinatX),
  406.                     $this->GetImageY($OldCoordinatY),
  407.                     $this->GetImageX($NewCoordinatX),
  408.                     $this->GetImageY($NewCoordinatY),
  409.                     $Color);
  410.  
  411.                 //Get some OldCoordinat*
  412.                 $OldCoordinatX = $NewCoordinatX;
  413.                 $OldCoordinatY = $NewCoordinatY;
  414.             }
  415.  
  416.             //Draw label if it is enabled
  417.             if($S->EnableLabel)
  418.             {
  419.                 //Draw label
  420.                 imagestring($ImageResource,$S->LabelFont,5,$LabelY,"- ".$S->Label,$Color);
  421.  
  422.                 //Add Label height to next Label X position
  423.                 $LabelY += imagefontheight($S->LabelFont);
  424.             }
  425.         }
  426.     }

  454.     /**
  455.     *Generates short numbers
  456.     *
  457.     *Rewrites numbers into scientific notation, with a certain maximum length.<br>
  458.     *Example: ShortNumber(501000000) == 5.01e8
  459.     *
  460.     *@access private
  461.     *@param integer $Number The number you wish to shorten.
  462.     *@param integer $MaxLen The maximum length of the output default to 7.
  463.     *@return string Scientific notation of the given Number at a certain length.
  464.     */
  465.     function ShortNumber($Number, $MaxLen = 7)
  466.     {
  467.         //If $Number isn't too long return it as it is
  468.         if(strlen($Number)<=$MaxLen)
  469.         {
  470.             return $Number;
  471.         }else{
  472.             //Convert to scientific notation
  473.             $NSci = sprintf("%e",$Number);
  474.  
  475.             //Follwing hack prevents the function from showing too many decimals
  476.             $ArrNSci = split($NSci,"e");
  477.             return round($ArrNSci[0],$MaxLen-5) . "e" . $ArrNSci[1];
  478.         }
  479.     }
