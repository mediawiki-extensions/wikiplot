  173.           //Initialize $Parser and creat an XML Parser to use later on
  174.           $this->Parser = xml_parser_create();
  175.        //Initialize $Input and set it equal to $Data (XML from user)

  182.           $this->Separator = "<graph";
  183.        
  184.           //Set XML Parser to use it within object
  185.        xml_set_object($this->Parser, $this);
  186.        //Set up start and end element handlers for the parser
  187.        xml_set_element_handler($this->Parser, "OpenTag", "CloseTag");
  188.        //Set up character data handler for the parser
  189.        xml_set_character_data_handler($this->Parser, "GetCharData");
  190.             
  191.        //Call ExplodeInputData() to get the interested tags

  193.        
  194.        //Call Parse() to parse the $Input
  195.        $this->Parse($this->Input);
  196.         
  197.        //Free the XML parser to later use

  262.    {
  263.           //Split the given XML data by using $Separator
  264.           $InterestedTags = explode($this->Separator , $this->Input); 
  265.           
  266.           //Go through the array containing the interesting tags
  267.           //NOTICE: $i must be = 1 because the array contains
  268.           //nothing on 0 position
  269.           //NOTICE: $ must be < lenght of the array and not <= because
  270.           //the last element in the array is not interesting.
  271.           for ($i=1; $i < count($InterestedTags); $i++)
  272.           {
  273.                  //Put the $Separator into the tag
  274.                  //(the separator vanishes when exploding the data)
  275.                  //fx. If the separator is <tag. The following will take place.
  276.                  //<tag>Hello</tag> will be exploded by <tag and
  277.                  //returned as >Hello</tag>. To complete the tag
  278.                  //we put the separator back on place. <tag + >Hello</tag>
  279.                  //this will return the complete tag = <tag>Hello</tag>
  280.                  array_push($this->Tags, $this->Separator . $InterestedTags[$i]);
  281.           }
  282.    }
